{{- $image := .Params.image -}} {{- if reflect.IsSlice $image }}{{ $image = index $image 0 }}{{ end -}} {{- with $image -}} {{- $originalURL := . | relURL -}} {{- $alt := default $.Title $.Params.image_alt -}} {{- $resourcePath := . | strings.TrimPrefix "/" -}} {{- $extension := path.Ext $resourcePath -}} {{- $masterPath := printf "image-sources/%s.png" (strings.TrimSuffix $extension $resourcePath) -}} {{- $source := resources.Get $masterPath -}} {{- if not $source -}} {{- $source = resources.Get $resourcePath -}} {{- end -}} {{- if $source -}} {{- $candidateWidths := slice -}} {{- range slice 400 768 980 -}} {{- if le . $source.Width -}} {{- $candidateWidths = $candidateWidths | append . -}} {{- end -}} {{- end -}} {{- if not (in $candidateWidths $source.Width) -}} {{- $candidateWidths = $candidateWidths | append $source.Width -}} {{- end -}} {{- $avifSrcset := slice -}} {{- $webpSrcset := slice -}} {{- range $candidateWidths -}} {{- $avif := $source.Resize (printf "%dx avif q58" .) -}} {{- $webp := $source.Resize (printf "%dx webp q78" .) -}} {{- $avifSrcset = $avifSrcset | append (printf "%s %dw" $avif.RelPermalink .) -}} {{- $webpSrcset = $webpSrcset | append (printf "%s %dw" $webp.RelPermalink .) -}} {{- end }} {{- else }} {{- end -}} {{- end -}}
